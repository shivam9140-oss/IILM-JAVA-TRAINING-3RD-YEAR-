OOPS basics Codes 
Code 1 
Class and Object creation

class Student {
    // logic
    int rollNumber;
    String name;
}
public class Main
{
	public static void main(String[] args) {
		Student obj = new Student();
		obj.rollNumber = 1;
		obj.name = "Abhishek";
		
		System.out.println(obj.rollNumber);
		System.out.println(obj.name);
	}
}

Code 2
Constructors ((parameterless)

class Student {
    // logic
    int rollNumber;
    String name;


    Student() {
        rollNumber = 1;
        name = "Abhishek";
    }
}

public class Main
{
	public static void main(String[] args) {
		Student obj = new Student();
		
		System.out.println(obj.rollNumber);
		System.out.println(obj.name);
	}
}

Code 3
Constructors ((parameterized)

class Student {
    // logic
    int rollNumber;
    String name;


    Student(int i, String n) {
        rollNumber = i;
        name = n;
    }
}

public class Main
{
	public static void main(String[] args) {
		Student obj = new Student(1,"Abhishek");
		
		System.out.println(obj.rollNumber);
		System.out.println(obj.name);
	}
}

Code 4
Attributes (self vs local variables)

class Student {
    
    int rollNumber; // instance variable

    void setId(int rollNumber) { // local variable 
        this.rollNumber = rollNumber; 
    }
}

public class Main
{
	public static void main(String[] args) {
		Student obj = new Student();
		obj.setId(1);
		
		System.out.println(obj.rollNumber);
	}
}
